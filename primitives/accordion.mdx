#### Example Code

```tsx
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from "atlr.kit";

const App = () => (
  <Accordion type={"single"} bordered={false} css={{ width: "400px" }}>
    <AccordionItem value={"item1"} css={{}}>
      <AccordionTrigger>Accordion One</AccordionTrigger>
      <AccordionContent>
        Deserunt culpa sunt eiusmod magna excepteur minim in id.
      </AccordionContent>
    </AccordionItem>

    <AccordionItem value={"item2"}>
      <AccordionTrigger>Accordion Two</AccordionTrigger>
      <AccordionContent>
        Deserunt culpa sunt eiusmod magna excepteur minim in id.
      </AccordionContent>
    </AccordionItem>
  </Accordion>
);
```

### Accordion Anatomy

<ParamField path="Accordion" type="root">
  The root element of the accordion.
</ParamField>

<ParamField path="AccordionItem" type="item">
  Contains all the parts of a collapsible section. Use `value` to identify the
  item.
</ParamField>

<ParamField path="AccordionTrigger" type="HTMLButtonElement">
  Toggles the collapsed state of its associated item.
</ParamField>

<ParamField path="AccordionContent" type="HTMLDivElement">
  Contains the collapsible content for an item.
</ParamField>

## Example Code

```tsx
import {
  RadioGroup,
  RadioGroupItem,
  RadioGroupIndicator,
  RadioLabel,
} from "atlr.kit";

const App = () => {
  return (
    <RadioGroup>
      <RadioGroupItem value="light">
        <RadioLabel>Light</RadioLabel>
        <RadioGroupIndicator />
      </RadioGroupItem>

      <RadioGroupItem value="dark">
        <RadioLabel>Dark</RadioLabel>
        <RadioGroupIndicator />
      </RadioGroupItem>

      <RadioGroupItem value="auto">
        <RadioLabel>Auto</RadioLabel>
        <RadioGroupIndicator />
      </RadioGroupItem>
    </RadioGroup>
  );
};
```

```tsx tg.tsx

const App = () => {
  return (
    <ToggleGroupPrimitive {...args}>
      <ToggleGroupItem value="auto">Auto</ToggleGroupItem>
      <ToggleGroupItem value="light">Light</ToggleGroupItem>
      <ToggleGroupItem value="dark">Dark</ToggleGroupItem>
    </ToggleGroupPrimitive>
  );
};
```

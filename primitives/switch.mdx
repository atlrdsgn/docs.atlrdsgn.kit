### Example

```tsx switch.tsx
import {Switch, SwitchThumb} from '@atlr/kit'

const App = () => {
  const [checked, setChecked] = React.useState(false)

  return (
    <form>
      <Switch checked={checked} onChange={setChecked}>
        <SwitchThumb />
      </Switch>
    </form>
  )
}
```

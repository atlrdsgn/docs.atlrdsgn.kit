#### Example Code

```tsx
import { Appbar, AppbarSpot } from "atlr.kit";

const App = () => {
  return (
    <Appbar size="1" glass border>
      <AppbarSpot alignment="start">start</AppbarSpot>
      <AppbarSpot alignment="center">center</AppbarSpot>
      <AppbarSpot alignment="end">end</AppbarSpot>
    </Appbar>
  );
};
```

#### Appbar Props

<ParamField path="size" type="string | '1' | '2' | '3'  ">
....

```tsx
<Appbar size="1">..</Appbar>
```

</ParamField>

<ParamField path="glass" type="boolean | default={true}">
Gives the appbar a blurred glass effect.

```tsx
<Appbar glass>..</Appbar>
```

</ParamField>

<ParamField path="css" type="style prop">
Custom CSS prop, to apply your own custom styles.

```tsx
<Appbar css={{ paddingLeft: 10 }}>....</Appbar>
```

</ParamField>

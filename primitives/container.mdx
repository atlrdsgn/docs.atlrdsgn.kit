#### Example Code

```tsx button.tsx
import { Container } from "atlr.kit";

const App = () => (
  <Container size="sm" align="center">
    Proident reprehenderit proident labore.
  </Container>
);
```

#### Container Props

<ParamField path="size" type="string | 'sm' | 'md' | 'lg'">
  The max width of the container.

```tsx
<Container size="sm">....</Container>
```

</ParamField>

<ParamField path="align" type="'left' | 'center' | 'right' | string">
  Alignment of the content inside the container.

```tsx
<Container align="center">....</Container>
```

</ParamField>

<ParamField path="justify" type="'start' | 'center' | 'end' | string">
  Justifying of the content inside the container.

```tsx
<Container align="center">....</Container>
```

</ParamField>

<ParamField path="optimal" type="boolean">
.....

```tsx
<Container optimal>....</Container>
```

</ParamField>

<ParamField path="css" type="style prop">
Custom CSS prop, to apply your own custom styles.

```tsx
<Container css={{ height: "auto" }}>....</Container>
```

</ParamField>
